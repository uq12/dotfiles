#[cheat:fmt]
# To format lines, 50 words in each line:
cat <file> | fmt -w 50
#[tldr:fmt]
# fmt
# Reformat a text file by joining its paragraphs and limiting the line width to given number of characters (75 by default).
# More information: <https://www.gnu.org/software/coreutils/fmt>.

# Переформатировать файл:
fmt path/to/file

# Переформатировать файл, создав выходные строки из
(не более) `n` символов: 
fmt -w n path/to/file

# Переформатировать файл, не соединяя вместе строки
короче заданной ширины:
fmt -s path/to/file

# Переформатируйте файл с одинаковым интервалом (1
  пробел между словами и 2 пробела между абзацами):
  fmt -u path/to/file
